### Loading Data

```javascript
// Load JSON data
import facilitiesData from '../scripts/ghana_waste_facilities.json';

// Filter by region
const accraFacilities = facilitiesData.facilities.filter(
  f => f.region === 'Greater Accra'
);

// Filter by type
const landfills = facilitiesData.facilities.filter(
  f => f.center_type.includes('Landfill')
);

// Find nearest facility (using coordinates)
const findNearest = (userLat, userLng, facilities) => {
  return facilities.reduce((nearest, facility) => {
    const dist = haversineDistance(
      userLat, userLng, 
      facility.latitude, facility.longitude
    );
    return dist < nearest.distance 
      ? { facility, distance: dist } 
      : nearest;
  }, { facility: null, distance: Infinity });
};
```
